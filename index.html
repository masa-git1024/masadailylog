Google スプレッドシート連携
設定手順
  1. Google スプレッドシートを新規作成
  2. 「拡張機能」→「Apps Script」を開く
  3. 下のコードをすべて貼り付けて保存
  4. 「デプロイ」→「新しいデプロイ」を選択
  5. 種類: ウェブアプリ / アクセス: 全員 に設定
  6. 発行されたURLを上欄に貼り付けて保存
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // シートの取得または作成
  let sheet = ss.getSheetByName("ログ");
  if (!sheet) {
    sheet = ss.insertSheet("ログ");
    // ヘッダー行を設定
    const headers = [
      "記録日", "体重(kg)", "飲酒強度", "飲酒強度ラベル",
      "飲み方", "ラーメン", "セルフケア",
      "運動時間(分)", "運動内容", "学習時間(分)",
      "メモ", "Claudeアドバイス", "保存日時"
    ];
    sheet.appendRow(headers);

    // ヘッダー書式設定
    const headerRange = sheet.getRange(1, 1, 1, headers.length);
    headerRange.setBackground("#c4704a");
    headerRange.setFontColor("#ffffff");
    headerRange.setFontWeight("bold");
    headerRange.setFontSize(11);
    sheet.setFrozenRows(1);
  }

  const d = JSON.parse(e.postData.contents);
  const alcLabels = ["なし","少し(1〜2杯)","普通(3〜4杯)","多め(5〜6杯)","たくさん"];

  const row = [
    d.date,
    d.weight ? Number(d.weight) : "",
    d.alcoholLevel ? Number(d.alcoholLevel) : 0,
    alcLabels[d.alcoholLevel] || "なし",
    d.alcoholType || "",
    d.ramen ? "○" : "×",
    d.selfcare ? "○" : "×",
    d.exerciseTime ? Number(d.exerciseTime) : "",
    d.exerciseContent || "",
    d.studyTime ? Number(d.studyTime) : "",
    d.memo || "",
    d.aiAdvice || "",
    new Date().toLocaleString("ja-JP", {timeZone: "Asia/Tokyo"})
  ];

  sheet.appendRow(row);

  // 最新行に書式を適用
  const lastRow = sheet.getLastRow();

  // 体重列に数値フォーマット
  sheet.getRange(lastRow, 2).setNumberFormat("0.0");

  // 飲酒強度に応じて色付け
  const alcLevel = Number(d.alcoholLevel);
  const alcColors = ["#edf4ee","#fdf8e8","#fdf2e4","#fdf0e8","#fde8e8"];
  sheet.getRange(lastRow, 3, 1, 2).setBackground(alcColors[alcLevel] || "#ffffff");

  // ラーメン・セルフケアのチェックをセンタリング
  sheet.getRange(lastRow, 6, 1, 2).setHorizontalAlignment("center");

  // 行の高さを調整
  sheet.setRowHeight(lastRow, 28);

  // メモとアドバイスは折り返し設定
  sheet.getRange(lastRow, 11, 1, 2).setWrap(true);

  // 列幅の調整（初回のみ）
  if (lastRow === 2) {
    sheet.setColumnWidth(1, 100);   // 日付
    sheet.setColumnWidth(2, 80);    // 体重
    sheet.setColumnWidth(3, 70);    // 飲酒強度数値
    sheet.setColumnWidth(4, 120);   // 飲酒強度ラベル
    sheet.setColumnWidth(5, 90);    // 飲み方
    sheet.setColumnWidth(6, 80);    // ラーメン
    sheet.setColumnWidth(7, 90);    // セルフケア
    sheet.setColumnWidth(8, 90);    // 運動時間
    sheet.setColumnWidth(9, 150);   // 運動内容
    sheet.setColumnWidth(10, 90);   // 学習時間
    sheet.setColumnWidth(11, 220);  // メモ
    sheet.setColumnWidth(12, 300);  // アドバイス
    sheet.setColumnWidth(13, 160);  // 保存日時
  }

  return ContentService
    .createTextOutput(JSON.stringify({status: "ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput("Daily Log API is running.")
    .setMimeType(ContentService.MimeType.TEXT);
}
